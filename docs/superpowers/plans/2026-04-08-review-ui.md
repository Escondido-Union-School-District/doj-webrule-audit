# Review UI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a local web-based review interface that replaces CLI review commands with a clickable spreadsheet grid.

**Architecture:** Express.js server serves a single-page HTML app with vanilla JS. The server exposes REST API endpoints that read/write directly to the existing SQLite database via the `db.ts` module. Frontend renders a spreadsheet grid and auto-saves changes via fetch calls.

**Tech Stack:** Express.js, better-sqlite3 (existing), vanilla HTML/CSS/JS frontend, tsx for dev server

### Task 3: API — GET /api/pages

**Files:**
- Modify: `src/server/api.ts`

This is the main data endpoint. It returns paginated pages with all 15 check results per page, supporting filter query params.

- [ ] **Step 1: Implement GET /api/pages**

Replace the contents of `src/server/api.ts`:

```typescript
import { Router } from 'express';
import { getDb } from '../db.js';
import { CHECKS } from '../config.js';

export const apiRouter = Router();

apiRouter.get('/health', (_req, res) => {
  res.json({ status: 'ok' });
});

// GET /api/pages — paginated pages with check results
apiRouter.get('/pages', (req, res) => {
  const db = getDb();

  const page = Math.max(1, parseInt(req.query.page as string) || 1);
  const perPage = Math.min(50, Math.max(1, parseInt(req.query.perPage as string) || 10));
  const statusFilter = req.query.status as string || '';
  const siteFilter = req.query.site as string || '';
  const checkFilter = parseInt(req.query.check as string) || 0;
  const searchFilter = (req.query.search as string || '').trim();

  // Get the latest run ID (excluding imports)
  const latestRun = db.prepare(
    "SELECT id FROM audit_runs WHERE id NOT LIKE 'excel%' ORDER BY started_at DESC LIMIT 1"
  ).get() as { id: string } | undefined;

  const runId = latestRun?.id || '';

  // Build the filtered page query
  let countSql = 'SELECT COUNT(DISTINCT p.id) as total FROM pages p';
  let dataSql = 'SELECT DISTINCT p.id, p.page_name, p.url, p.site FROM pages p';

  const joins: string[] = [];
  const wheres: string[] = ['p.active = 1'];
  const params: (string | number)[] = [];

  // If filtering by status or check, join audit_results
  if ((statusFilter || checkFilter) && runId) {
    joins.push('LEFT JOIN audit_results ar ON ar.page_id = p.id AND ar.run_id = ?');
    params.push(runId);

    if (checkFilter) {
      joins[0] = joins[0].replace('LEFT JOIN', 'JOIN');
      wheres.push('ar.check_number = ?');
      params.push(checkFilter);
    }

    if (statusFilter === 'unreviewed') {
      wheres.push("(ar.status = 'needs-review' OR ar.status IS NULL)");
    } else if (statusFilter === 'pass') {
      wheres.push("ar.status = 'pass'");
    } else if (statusFilter === 'fail') {
      wheres.push("ar.status = 'fail'");
    }
  }

  if (siteFilter) {
    wheres.push('p.site = ?');
    params.push(siteFilter);
  }

  if (searchFilter) {
    wheres.push('(p.page_name LIKE ? OR p.url LIKE ?)');
    params.push(`%${searchFilter}%`, `%${searchFilter}%`);
  }

  const joinClause = joins.join(' ');
  const whereClause = wheres.length ? ' WHERE ' + wheres.join(' AND ') : '';

  // Get total count
  const total = (db.prepare(`${countSql} ${joinClause} ${whereClause}`).get(...params) as { total: number }).total;

  // Get page IDs for current page
  const offset = (page - 1) * perPage;
  const pageRows = db.prepare(
    `${dataSql} ${joinClause} ${whereClause} ORDER BY p.site, p.page_name LIMIT ? OFFSET ?`
  ).all(...params, perPage, offset) as Array<{ id: number; page_name: string; url: string; site: string }>;

  // Get all check results for these pages
  const pageIds = pageRows.map(p => p.id);
  let resultsMap: Record<number, Record<number, { status: string; notes: string }>> = {};

  if (pageIds.length > 0 && runId) {
    const placeholders = pageIds.map(() => '?').join(',');
    const results = db.prepare(`
      SELECT page_id, check_number, 
        COALESCE(manual_override, status) as status,
        notes
      FROM audit_results
      WHERE page_id IN (${placeholders}) AND run_id = ?
    `).all(...pageIds, runId) as Array<{
      page_id: number; check_number: number; status: string; notes: string;
    }>;

    for (const r of results) {
      if (!resultsMap[r.page_id]) resultsMap[r.page_id] = {};
      resultsMap[r.page_id][r.check_number] = {
        status: r.status === 'needs-review' ? 'unreviewed' : r.status,
        notes: r.notes || '',
      };
    }
  }

  // Build response
  const data = pageRows.map(p => ({
    id: p.id,
    pageName: p.page_name,
    url: p.url,
    site: p.site,
    checks: Object.fromEntries(
      CHECKS.map(c => [
        c.number,
        resultsMap[p.id]?.[c.number] || { status: 'unreviewed', notes: '' },
      ])
    ),
  }));

  res.json({
    pages: data,
    pagination: { page, perPage, total, totalPages: Math.ceil(total / perPage) },
  });
});
```

// PATCH /api/results/:pageId/:checkNumber — update a single check
apiRouter.patch('/results/:pageId/:checkNumber', (req, res) => {
  const db = getDb();
  const pageId = parseInt(req.params.pageId, 10);
  const checkNumber = parseInt(req.params.checkNumber, 10);
  const { status, notes } = req.body;

  const latestRun = db.prepare(
    "SELECT id FROM audit_runs WHERE id NOT LIKE 'excel%' ORDER BY started_at DESC LIMIT 1"
  ).get() as { id: string } | undefined;

  if (!latestRun) {
    res.status(400).json({ error: 'No audit run found' });
    return;
  }

  const checkInfo = CHECKS.find(c => c.number === checkNumber);
  if (!checkInfo) {
    res.status(400).json({ error: 'Invalid check number' });
    return;
  }

  // Upsert: update if exists, insert if not
  const existing = db.prepare(
    'SELECT id FROM audit_results WHERE page_id = ? AND check_number = ? AND run_id = ?'
  ).get(pageId, checkNumber, latestRun.id);

  if (existing) {
    db.prepare(`
      UPDATE audit_results
      SET manual_override = ?, status = ?, notes = ?, audited_by = 'manual', audit_date = datetime('now')
      WHERE page_id = ? AND check_number = ? AND run_id = ?
    `).run(status, status, notes ?? null, pageId, checkNumber, latestRun.id);
  } else {
    db.prepare(`
      INSERT INTO audit_results (page_id, check_number, check_name, status, manual_override, notes, audited_by, run_id)
      VALUES (?, ?, ?, ?, ?, ?, 'manual', ?)
    `).run(pageId, checkNumber, checkInfo.name, status, status, notes ?? null, latestRun.id);
  }

  // Also update manual_queue if there's a pending entry
  db.prepare(`
    UPDATE manual_queue
    SET status = 'done', result = ?, notes = ?, reviewer = 'web-ui', reviewed_at = datetime('now')
    WHERE page_id = ? AND check_number = ? AND status = 'pending'
  `).run(status, notes ?? null, pageId, checkNumber);

  res.json({ ok: true });
});
```

// POST /api/results/:pageId/pass-all — pass all unreviewed checks for a page
apiRouter.post('/results/:pageId/pass-all', (req, res) => {
  const db = getDb();
  const pageId = parseInt(req.params.pageId, 10);

  const latestRun = db.prepare(
    "SELECT id FROM audit_runs WHERE id NOT LIKE 'excel%' ORDER BY started_at DESC LIMIT 1"
  ).get() as { id: string } | undefined;

  if (!latestRun) {
    res.status(400).json({ error: 'No audit run found' });
    return;
  }

  // Find checks that are needs-review or missing
  const existing = db.prepare(`
    SELECT check_number, status, manual_override
    FROM audit_results
    WHERE page_id = ? AND run_id = ?
  `).all(pageId, latestRun.id) as Array<{ check_number: number; status: string; manual_override: string | null }>;

  const existingMap = new Map(existing.map(r => [r.check_number, r]));
  let updated = 0;

  for (const check of CHECKS) {
    const result = existingMap.get(check.number);
    const effectiveStatus = result?.manual_override || result?.status;

    // Only update if unreviewed (needs-review, or no result)
    if (!result || effectiveStatus === 'needs-review') {
      if (result) {
        db.prepare(`
          UPDATE audit_results
          SET manual_override = 'pass', status = 'pass', audited_by = 'manual-batch', audit_date = datetime('now')
          WHERE page_id = ? AND check_number = ? AND run_id = ?
        `).run(pageId, check.number, latestRun.id);
      } else {
        db.prepare(`
          INSERT INTO audit_results (page_id, check_number, check_name, status, manual_override, audited_by, run_id)
          VALUES (?, ?, ?, 'pass', 'pass', 'manual-batch', ?)
        `).run(pageId, check.number, check.name, latestRun.id);
      }

      // Clear manual queue
      db.prepare(`
        UPDATE manual_queue
        SET status = 'done', result = 'pass', reviewer = 'web-ui', reviewed_at = datetime('now')
        WHERE page_id = ? AND check_number = ? AND status = 'pending'
      `).run(pageId, check.number);

      updated++;
    }
  }

  // Return updated check states
  const updatedResults = db.prepare(`
    SELECT check_number, COALESCE(manual_override, status) as status, notes
    FROM audit_results
    WHERE page_id = ? AND run_id = ?
  `).all(pageId, latestRun.id) as Array<{ check_number: number; status: string; notes: string }>;

  const checks: Record<number, { status: string; notes: string }> = {};
  for (const r of updatedResults) {
    checks[r.check_number] = {
      status: r.status === 'needs-review' ? 'unreviewed' : r.status,
      notes: r.notes || '',
    };
  }

  res.json({ ok: true, updated, checks });
});
```

  const params = new URLSearchParams({
    page: currentPage,
    perPage: perPage,
  });
  if (filters.status) params.set('status', filters.status);
  if (filters.site) params.set('site', filters.site);
  if (filters.check) params.set('check', filters.check);
  if (filters.search) params.set('search', filters.search);

  const res = await fetch(`/api/pages?${params}`);
  const data = await res.json();

  if (data.pages.length === 0) {
    container.innerHTML = '<div class="loading">No pages match your filters.</div>';
    document.getElementById('pagination').style.display = 'none';
    return;
  }

  container.innerHTML = '';

  for (const page of data.pages) {
    container.appendChild(buildPageTable(page));
  }

  // Update pagination
  const pag = data.pagination;
  document.getElementById('pagination').style.display = 'flex';
  const start = (pag.page - 1) * pag.perPage + 1;
  const end = Math.min(pag.page * pag.perPage, pag.total);
  document.getElementById('page-info').textContent = `Showing ${start}–${end} of ${pag.total} pages`;
  document.getElementById('btn-prev').disabled = pag.page <= 1;
  document.getElementById('btn-next').disabled = pag.page >= pag.totalPages;
}

// --- Build a table for one page ---
function buildPageTable(page) {
  const wrapper = document.createElement('div');
  wrapper.style.overflowX = 'auto';
  wrapper.style.marginBottom = '1.5em';

  const table = document.createElement('table');
  table.className = 'review-table';

  // --- Header row 1 (checks 1-8) ---
  const thead = document.createElement('thead');
  const headerRow = document.createElement('tr');

  const pageHeader = document.createElement('th');
  pageHeader.className = 'page-hdr';
  pageHeader.textContent = 'Page';
  headerRow.appendChild(pageHeader);

  for (const cn of ROW1_CHECKS) {
    const th = document.createElement('th');
    th.className = 'check-start';
    th.colSpan = 2;
    th.textContent = CHECK_LABELS[cn];
    headerRow.appendChild(th);
  }
  thead.appendChild(headerRow);
  table.appendChild(thead);

  // --- Data row 1 (checks 1-8) ---
  const tbody = document.createElement('tbody');

  const dataRow1 = document.createElement('tr');
  const pageCell = document.createElement('td');
  pageCell.className = 'page-cell';
  pageCell.rowSpan = 3;

  const link = document.createElement('a');
  link.href = page.url;
  link.target = '_blank';
  link.textContent = page.pageName;
  pageCell.appendChild(link);

  const urlDiv = document.createElement('div');
  urlDiv.className = 'url';
  urlDiv.textContent = page.url.replace(/^https?:\/\//, '');
  pageCell.appendChild(urlDiv);

  const passAllBtn = document.createElement('button');
  passAllBtn.className = 'pass-all-btn';
  passAllBtn.textContent = '✓ Pass All';
  passAllBtn.addEventListener('click', () => handlePassAll(page.id, table));
  pageCell.appendChild(passAllBtn);

  dataRow1.appendChild(pageCell);
  appendCheckCells(dataRow1, page, ROW1_CHECKS, false);
  tbody.appendChild(dataRow1);

  // --- Header row 2 (checks 9-15) ---
  const headerRow2 = document.createElement('tr');
  for (const cn of ROW2_CHECKS) {
    const th = document.createElement('th');
    th.className = 'check-start';
    th.colSpan = 2;
    th.textContent = CHECK_LABELS[cn];
    headerRow2.appendChild(th);
  }
  tbody.appendChild(headerRow2);

  // --- Data row 2 (checks 9-15) ---
  const dataRow2 = document.createElement('tr');
  dataRow2.className = 'row2 page-sep';
  appendCheckCells(dataRow2, page, ROW2_CHECKS, true);
  tbody.appendChild(dataRow2);

  table.appendChild(tbody);
  wrapper.appendChild(table);
  return wrapper;
}

// --- Append P/F + notes cells for a set of checks ---
function appendCheckCells(row, page, checkNumbers, isRow2) {
  for (let i = 0; i < checkNumbers.length; i++) {
    const cn = checkNumbers[i];
    const isOdd = cn % 2 === 1;
    const checkData = page.checks[cn] || { status: 'unreviewed', notes: '' };

    // P/F cell
    const pfCell = document.createElement('td');
    pfCell.className = `pf-cell ${isOdd ? 'check-odd' : 'check-even'} check-start`;
    if (isRow2) pfCell.classList.add('row2');

    const pfDiv = document.createElement('div');
    pfDiv.className = `pf pf-${checkData.status}`;
    pfDiv.textContent = statusToLabel(checkData.status);
    pfDiv.dataset.pageId = page.id;
    pfDiv.dataset.check = cn;
    pfDiv.dataset.status = checkData.status;
    pfDiv.addEventListener('click', handlePfClick);
    pfCell.appendChild(pfDiv);
    row.appendChild(pfCell);

    // Notes cell
    const noteCell = document.createElement('td');
    noteCell.className = `note-cell ${isOdd ? 'check-odd' : 'check-even'}`;
    if (isRow2) noteCell.classList.add('row2');

    const textarea = document.createElement('textarea');
    textarea.className = 'note-input';
    textarea.value = checkData.notes;
    textarea.dataset.pageId = page.id;
    textarea.dataset.check = cn;
    textarea.addEventListener('blur', handleNoteBlur);
    noteCell.appendChild(textarea);
    row.appendChild(noteCell);
  }
}

// --- Status helpers ---
function statusToLabel(status) {
  if (status === 'pass') return 'P';
  if (status === 'fail') return 'F';
  return '?';
}

function cycleStatus(current) {
  if (current === 'unreviewed') return 'pass';
  if (current === 'pass') return 'fail';
  return 'unreviewed';
}

// --- Event handlers ---
async function handlePfClick(e) {
  const div = e.currentTarget;
  const pageId = div.dataset.pageId;
  const checkNumber = div.dataset.check;
  const newStatus = cycleStatus(div.dataset.status);

  // Update UI immediately
  div.dataset.status = newStatus;
  div.className = `pf pf-${newStatus}`;
  div.textContent = statusToLabel(newStatus);

  // Find the notes textarea for this check
  const noteCell = div.closest('tr').querySelector(`textarea[data-check="${checkNumber}"]`);
  const notes = noteCell ? noteCell.value : '';

  // Save to server
  await fetch(`/api/results/${pageId}/${checkNumber}`, {
    method: 'PATCH',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ status: newStatus === 'unreviewed' ? 'needs-review' : newStatus, notes }),
  });
}

async function handleNoteBlur(e) {
  const textarea = e.currentTarget;
  const pageId = textarea.dataset.pageId;
  const checkNumber = textarea.dataset.check;
  const notes = textarea.value;

  // Find the P/F div for this check to get current status
  const row = textarea.closest('tr');
  const pfDiv = row.querySelector(`.pf[data-check="${checkNumber}"]`);
  const status = pfDiv ? pfDiv.dataset.status : 'unreviewed';

  await fetch(`/api/results/${pageId}/${checkNumber}`, {
    method: 'PATCH',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ status: status === 'unreviewed' ? 'needs-review' : status, notes }),
  });
}

async function handlePassAll(pageId, table) {
  const res = await fetch(`/api/results/${pageId}/pass-all`, { method: 'POST' });
  const data = await res.json();

  if (data.ok) {
    // Update all P/F cells in this table
    const pfDivs = table.querySelectorAll('.pf');
    for (const div of pfDivs) {
      const cn = parseInt(div.dataset.check);
      const checkResult = data.checks[cn];
      if (checkResult) {
        div.dataset.status = checkResult.status;
        div.className = `pf pf-${checkResult.status}`;
        div.textContent = statusToLabel(checkResult.status);
      }
    }
  }
}

// --- Filter events ---
function bindFilterEvents() {
  document.getElementById('filter-status').addEventListener('change', (e) => {
    filters.status = e.target.value;
    currentPage = 1;
    loadPages();
  });
  document.getElementById('filter-site').addEventListener('change', (e) => {
    filters.site = e.target.value;
    currentPage = 1;
    loadPages();
  });
  document.getElementById('filter-check').addEventListener('change', (e) => {
    filters.check = e.target.value;
    currentPage = 1;
    loadPages();
  });
  document.getElementById('filter-search').addEventListener('input', (e) => {
    clearTimeout(debounceTimer);
    debounceTimer = setTimeout(() => {
      filters.search = e.target.value;
      currentPage = 1;
      loadPages();
    }, 300);
  });
}

// --- Pagination events ---
function bindPaginationEvents() {
  document.getElementById('btn-prev').addEventListener('click', () => {
    if (currentPage > 1) { currentPage--; loadPages(); }
  });
  document.getElementById('btn-next').addEventListener('click', () => {
    currentPage++;
    loadPages();
  });
  document.getElementById('per-page').addEventListener('change', (e) => {
    perPage = parseInt(e.target.value);
    currentPage = 1;
    loadPages();
  });
}
